# Function to split sentences using a regular expression
def extract_sentences(text):
    """
    Extract sentences from a given text using the regex [^\.\!\?]*[\.\!\?]
    :param text: str, input abstract text
    :return: list of sentences
    """
    sentence_pattern = "[^\.\!\?]*[\.\!\?]"
    sentences = re.findall(sentence_pattern, text)
    return [s.strip() for s in sentences if s.strip()]

# Main script
def count_sentences_in_abstracts(file_path):
    """
    Count the number of sentences in each abstract in the given TSV file.
    :param file_path: str, path to the TSV file
    :return: list of sentence counts for each abstract
    """
    sentence_counts = []
    with open(file_path, "r", encoding="utf-8") as file:
        for line in file:
            # Split the line into columns (PMID, title, abstract)
            columns = line.strip().split("\t")
            abstract = columns[2] # Get the abstract
            sentences = extract_sentences(abstract)
            sentence_counts.append(len(sentences))
    return sentence_counts
